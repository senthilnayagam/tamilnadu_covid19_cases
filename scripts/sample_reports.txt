# death per district

SELECT 
    district,
    count(district)
FROM 
    Cases
GROUP BY 
    district
ORDER BY 
    count(district) DESC;



# age mortality, top count

SELECT 
    age,
    count(age)
FROM 
    Cases
GROUP BY 
    age
ORDER BY 
    count(age) DESC, age DESC;


# death by age

SELECT 
    age,
    count(age)
FROM 
    Cases
GROUP BY 
    age
ORDER BY 
     age DESC;
